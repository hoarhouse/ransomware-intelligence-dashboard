        // API Status Management

        function updateAPIStatus(elementId, success, responseTime = null) {

            const element = document.getElementById(elementId);

            if (element) {

                element.className = success ? 'api-status api-success' : 'api-status api-error';

                element.textContent = success ? 

                    (responseTime ? `● ${responseTime}ms` : '● LIVE') : 

                    '● ERROR';

            }

        }


        // Main API call to serverless function

        async function fetchRansomwareData() {

            const startTime = Date.now();

            

            try {

                console.log('🔍 Fetching data from serverless API...');

                const response = await fetch('/.netlify/functions/ransomware-api', {

                    method: 'GET',

                    headers: {

                        'Accept': 'application/json',

                        'Content-Type': 'application/json'

                    }

                });


                const responseTime = Date.now() - startTime;

                console.log(`⚡ API Response time: ${responseTime}ms`);


                if (!response.ok) {

                    throw new Error(`HTTP ${response.status}: ${response.statusText}`);

                }


                const data = await response.json();

                console.log('✅ Serverless API data received:', data);


                // Update API metrics

                updateAPIMetrics(data, responseTime);


                // Update dashboard with real data

                updateDashboardWithServerlessData(data);


                return data;


            } catch (error) {

                console.error('❌ Serverless API Error:', error);

                // Fall back to simulated data

                const simulatedData = generateFallbackData();

                updateDashboardWithServerlessData(simulatedData);

                return simulatedData;

            }

        }

        function updateDashboardWithServerlessData(data) {

            const { ransomwhere, cisa, canadian } = data.sources;


            // Update status indicators

            updateAPIStatus('ransomwhereStatus', ransomwhere.status === 'success');

            updateAPIStatus('cisaStatus', cisa.status === 'success');

            updateAPIStatus('canadaStatus', canadian.status === 'success');


            // Update viral stats

            if (ransomwhere.status === 'success' && ransomwhere.data) {

                const totalPayments = ransomwhere.data.reduce((sum, payment) => sum + payment.amount, 0);

                document.getElementById('ransomPayments').textContent = totalPayments.toFixed(2) + ' BTC';

                document.getElementById('ransomwarePayments').textContent = ransomwhere.data.length;

            } else {

                document.getElementById('ransomPayments').textContent = 'API Error';

                document.getElementById('ransomwarePayments').textContent = '---';

            }


            if (cisa.status === 'success' && cisa.data) {

                const ransomwareAlerts = cisa.data.filter(alert => alert.isRansomware).length;

                document.getElementById('cisaAlerts').textContent = ransomwareAlerts;

                document.getElementById('recentAlerts').textContent = cisa.data.length;

            } else {

                document.getElementById('cisaAlerts').textContent = 'API Error';

                document.getElementById('recentAlerts').textContent = '---';

            }


            if (canadian.status === 'success' && canadian.data) {

                const recentAlerts = canadian.data.filter(alert => alert.isRecent).length;

                document.getElementById('canadianAlerts').textContent = recentAlerts;

            } else {

                document.getElementById('canadianAlerts').textContent = '---';

            }


            // Update sections with real data

            updateRansomwhereSection(ransomwhere);

            updateCISASection(cisa);

            updateCanadianSection(canadian);

            updatePaymentAnalysis(ransomwhere);


            // Generate remaining simulated data

            generateSimulatedSupportingData();

        }


        function updateRansomwhereSection(ransomwhere) {

            if (ransomwhere.status === 'success' && ransomwhere.data) {

                const familyStats = {};

                ransomwhere.data.forEach(payment => {

                    familyStats[payment.family] = (familyStats[payment.family] || 0) + payment.amount;

                });


                const sortedFamilies = Object.entries(familyStats)

                    .sort(([,a], [,b]) => b - a)

                    .slice(0, 8);


                document.getElementById('ransomwhereData').innerHTML = sortedFamilies

                    .map(([family, amount]) => `

                        <div class="list-item real-data">

                            <span class="item-name">${family}</span>

                            <span class="item-value">${amount.toFixed(4)} BTC</span>

                        </div>

                    `).join('');

            } else {

                document.getElementById('ransomwhereData').innerHTML = `

                    <div style="color: #dc2626; text-align: center; padding: 20px;">

                        API Error: ${ransomwhere.error || 'Unknown error'}

                    </div>

                `;

            }

        }
